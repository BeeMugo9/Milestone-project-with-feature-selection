# Defining a function that will check for overfitting in the train and test set

def check_overfitting(model, X_train, X_test, y_train, y_test):
    # Make predictions
    y_pred_train = model.predict(X_train)
    y_pred_test = model.predict(X_test)

    # Compute MAE
    train_mae = mean_absolute_error(y_train, y_pred_train)
    test_mae = mean_absolute_error(y_test, y_pred_test)

   # Compute RMSE (taking the square root manually)
    train_rmse = np.sqrt(mean_squared_error(y_train, y_pred_train))
    test_rmse = np.sqrt(mean_squared_error(y_test, y_pred_test))

    # Compute R-squared
    train_r2 = r2_score(y_train, y_pred_train)
    test_r2 = r2_score(y_test, y_pred_test)

    results = {
        'train_mae': train_mae,
        'test_mae': test_mae,
        'train_rmse': train_rmse,
        'test_rmse': test_rmse,
        'train_r2': train_r2,
        'test_r2': test_r2
    }

    return results

# Function to evaluate model on both train and test sets
def evaluate_model(y_true_train, y_pred_train, y_true_test, y_pred_test):
    # For training data
    mae_train = mean_absolute_error(y_true_train, y_pred_train)
    rmse_train = np.sqrt(mean_squared_error(y_true_train, y_pred_train))
    r2_train = r2_score(y_true_train, y_pred_train)

    # For testing data
    mae_test = mean_absolute_error(y_true_test, y_pred_test)
    rmse_test = np.sqrt(mean_squared_error(y_true_test, y_pred_test))
    r2_test = r2_score(y_true_test, y_pred_test)

    return mae_train, rmse_train, r2_train, mae_test, rmse_test, r2_test
